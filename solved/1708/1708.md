# 1708번: 볼록 껍질

- 그라함 스캔(Graham Scan)이라는 알고리즘을 활용한 문제!
- 어렵다,, 그래도 기하학에 대한 지식을 쌓아갈 수 있음에 감사하다!

```cpp
#include <bits/stdc++.h>
using namespace std;

struct vector2 {
    long long x, y;
    long long p, q;
    
    explicit vector2(long long x_ = 0, long long y_ = 0,
        long long p_ = 0, long long q_ = 0) : x(x_), y(y_), p(p_), q(q_) {}
    
    vector2 operator - (const vector2& rhs) const {
        return vector2(x - rhs.x, y - rhs.y);
    }
    
    bool operator < (const vector2& rhs) const {
        if (q * rhs.p != p * rhs.q) return q * rhs.p < p * rhs.q;
        return x != rhs.x ? x < rhs.x : y < rhs.y;
    }
};

int ccw(vector2 a, vector2 b) {
    long long res = (a.x * b.y - a.y * b.x);
    if (res > 0) return 1;
    if (res == 0) return 0;
    return -1;
}

int ccw(vector2 p, vector2 a, vector2 b) {
    return ccw(a - p, b - p);
}

int main(void) {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int N;
    cin >> N;
    
    vector<vector2> p(N);
    
    for (int i = 0; i < N; i++) {
        cin >> p[i].x >> p[i].y;
    }
    sort(p.begin(), p.end());
    
    vector2 pivot = p[0];
    for (int i = 1; i < N; i++) {
        p[i].q = p[i].y - pivot.y;
        p[i].p = p[i].x - pivot.x;
    }
    sort(p.begin() + 1, p.end());
    
    vector<vector2> hull;
    hull.push_back(pivot);
    hull.push_back(p[1]);
    
    int next = 2;
    while (next < p.size()) {
        while (hull.size() >= 2) {
            vector2 first, second;
            second = hull.back();
            hull.pop_back();
            first = hull.back();
            // first를 기준으로 p[next]가 second의 반시계 방향이라면
            if (ccw(first, second, p[next]) > 0) {
                hull.push_back(second); break;
            }
        }
        hull.push_back(p[next++]);
    }
    cout << hull.size() << "\n";
    
    return 0;
}
```
