```cpp
#include<iostream>
#include<string.h>
#include<math.h>
#include<vector>
using namespace std;
int P[21];
int D[1000001][30];
vector<int> v;

int go(int n, int before)
{

	if (n == 0)
	{
		return 1;
	}
	if (n < 0)
	{
		return 0;
	}

	if (D[n][before] > 0)
	{
		return D[n][before];
	}
    
	for (int i = 20; i>=0 ; i--)
	{
		if (i <= before && n-pow(2,i)>=0)
		{
			D[n][before] += go(n - pow(2,i), i)%1000000000;
			D[n][before] %= 1000000000;

		}
	}

	return D[n][before]%1000000000;
}

int main()
{
	int N;
	cin >> N;
	memset(D, 0, sizeof(D));

	printf("%d\n", go(N,20) % 1000000000);

	return 0;
}
```
